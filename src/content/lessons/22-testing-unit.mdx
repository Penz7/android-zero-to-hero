### Mockito cho Unit Tests

```kotlin
import org.mockito.kotlin.*

class UserViewModelTest {
    
    private lateinit var viewModel: UserViewModel
    private val mockUserRepository = mock<UserRepository>()
    private val mockGetUsersUseCase = mock<GetUsersUseCase>()
    
    @BeforeEach
    fun setup() {
        viewModel = UserViewModel(mockGetUsersUseCase)
    }
    
    @Test
    fun `loadUsers should emit Success when repository returns data`() = runTest {
        // Given
        val fakeUsers = listOf(
            User(1, "Minh", "minh@test.com"),
            User(2, "Tuấn", "tuan@test.com")
        )
        whenever(mockGetUsersUseCase()).thenReturn(flowOf(Resource.Success(fakeUsers)))
        
        // When
        viewModel.loadUsers()
        
        // Then
        assertEquals(
            UserUiState.Success(fakeUsers),
            viewModel.uiState.value
        )
    }
    
    @Test
    fun `loadUsers should emit Error when repository throws exception`() = runTest {
        // Given
        whenever(mockGetUsersUseCase()).thenReturn(
            flowOf(Resource.Error("Lỗi mạng"))
        )
        
        // When
        viewModel.loadUsers()
        
        // Then
        assertEquals(
            UserUiState.Error("Lỗi mạng"),
            viewModel.uiState.value
        )
    }
    
    @Test
    fun `deleteUser should call repository delete`() = runTest {
        // Given
        whenever(mockUserRepository.deleteUser(1)).thenReturn(Resource.Success(Unit))
        
        // When
        viewModel.deleteUser(1)
        
        // Then
        verify(mockUserRepository).deleteUser(1)
    }
}
```

### Testing Coroutines

```kotlin
import kotlinx.coroutines.test.runTest
import kotlinx.coroutines.test.TestDispatcher
import kotlinx.coroutines.test.UnconfinedTestDispatcher

class CoroutineTest {
    
    @Test
    fun `test suspend function`() = runTest {
        val result = suspendFunction()
        assertEquals("expected", result)
    }
    
    @Test
    fun `test flow emission`() = runTest {
        val flow = flow {
            emit(1)
            emit(2)
            emit(3)
        }
        
        val results = mutableListOf<Int>()
        flow.collect { results.add(it) }
        
        assertEquals(listOf(1, 2, 3), results)
    }
    
    @Test
    fun `test flow with Turbine`() = runTest {
        val flow = flowOf(1, 2, 3)
        
        flow.test {
            assertEquals(1, awaitItem())
            assertEquals(2, awaitItem())
            assertEquals(3, awaitItem())
            awaitComplete()
        }
    }
}
```

### Testing Use Cases

```kotlin
class GetUsersUseCaseTest {
    
    private lateinit var useCase: GetUsersUseCase
    private val mockRepository = mock<UserRepository>()
    
    @BeforeEach
    fun setup() {
        useCase = GetUsersUseCase(mockRepository)
    }
    
    @Test
    fun `invoke should return users from repository`() = runTest {
        // Given
        val expectedUsers = listOf(User(1, "Test", "test@test.com"))
        whenever(mockRepository.getUsers()).thenReturn(
            flowOf(Resource.Success(expectedUsers))
        )
        
        // When
        val result = useCase()
        
        // Then
        result.test {
            val loading = awaitItem()
            assertTrue(loading is Resource.Loading)
            
            val success = awaitItem()
            assertTrue(success is Resource.Success)
            assertEquals(expectedUsers, success.data)
            
            awaitComplete()
        }
    }
}
```

### Testing Repository

```kotlin
class UserRepositoryImplTest {
    
    private lateinit var repository: UserRepositoryImpl
    private val mockRemoteDataSource = mock<UserRemoteDataSource>()
    private val mockLocalDataSource = mock<UserLocalDataSource>()
    
    @BeforeEach
    fun setup() {
        repository = UserRepositoryImpl(mockRemoteDataSource, mockLocalDataSource)
    }
    
    @Test
    fun `getUsers should emit local data then remote`() = runTest {
        // Given
        val localUsers = listOf(UserEntity(1, "Local", "local@test.com"))
        val remoteUsers = listOf(UserDto(1, "Remote", "remote@test.com", null))
        
        whenever(mockLocalDataSource.getUsers()).thenReturn(flowOf(localUsers))
        whenever(mockRemoteDataSource.getUsers()).thenReturn(remoteUsers)
        
        // When & Then
        repository.getUsers().test {
            val loading = awaitItem()
            assertTrue(loading is Resource.Loading)
            
            val localData = awaitItem()
            assertTrue(localData is Resource.Success)
            assertEquals(1, localData.data?.size)
            
            awaitComplete()
        }
        
        verify(mockLocalDataSource).insertUsers(any())
    }
}
```

### Parameterized Tests

        "100, 200, 300"

        assertEquals(expected, Calculator.add(a, b))
    }
}

### Assertions trong Unit Tests

```kotlin
class AssertionTest {
    
    @Test
    fun `test various assertions`() {
        // Equality
        assertEquals(5, 5)
        assertNotEquals(5, 6)
        
        // Boolean
        assertTrue(true)
        assertFalse(false)
        
        // Null checks
        assertNotNull("not null")
        assertNull(null)
        
        // Collections
        val list = listOf(1, 2, 3)
        assertIterableEquals(listOf(1, 2, 3), list)
        assertThrows<NoSuchElementException> {
            emptyList<Int>().first()
        }
        
        // String
        val message = "Hello World"
        assertContains(message, "World")
        assertStartsWith(message, "Hello")
    }
}
```

### Mẹo thực tế

> **💡 Tip**: Đặt tên test mô tả rõ hành vi được test: `should [expected behavior] when [condition]`. Ví dụ: `should emit Error when network fails`.

> **⚠️ Lưu ý**: Không test implementation details, chỉ test behavior (đầu vào → đầu ra). Điều này giúp test ít bị break khi refactor code.

> **🎯 Best Practice**: Aim cho 70-80% code coverage. 100% coverage không có nghĩa là code không có bug, nhưng <50% là tín hiệu xấu.
