### ComposeTestRule

```kotlin
import androidx.compose.ui.test.junit4.createComposeRule
import androidx.compose.ui.test.*

class UserScreenTest {
    
    @get:Rule
    val composeTestRule = createComposeRule()
    
    @Test
    fun `user list displays users correctly`() {
        // Given
        val users = listOf(
            User(1, "Minh", "minh@test.com"),
            User(2, "Tuấn", "tuan@test.com")
        )
        
        // When
        composeTestRule.setContent {
            MaterialTheme {
                UserListScreen(users = users)
            }
        }
        
        // Then
        composeTestRule.onNodeWithText("Minh").assertExists()
        composeTestRule.onNodeWithText("Tuấn").assertExists()
        composeTestRule.onNodeWithText("minh@test.com").assertExists()
    }
}
```

### Finding Nodes

```kotlin
class FindingNodesTest {
    
    @get:Rule
    val composeTestRule = createComposeRule()
    
    @Test
    fun `find nodes by different matchers`() {
        composeTestRule.setContent {
            MaterialTheme {
                Column {
                    Text("Tiêu đề", style = MaterialTheme.typography.headlineMedium)
                    Button(onClick = {}) { Text("Nhấn") }
                    TextField(
                        value = "",
                        onValueChange = {},
                        label = { Text("Nhập tên") }
                    )
                }
            }
        }
        
        // Find by text
        composeTestRule.onNodeWithText("Tiêu đề").assertExists()
        
        // Find by content description
        composeTestRule.onNodeWithContentDescription("Nhấn").assertExists()
        
        // Find by test tag
        composeTestRule.onNodeWithTag("input_field").assertExists()
        
        // Find by semantics
        composeTestRule.onNode(hasText("Tiêu đề")).assertExists()
        composeTestRule.onNode(isSelectable()).assertExists()
    }
}
```

### Performing Actions

```kotlin
class ActionTest {
    
    @get:Rule
    val composeTestRule = createComposeRule()
    
    @Test
    fun `button click increments counter`() {
        var count = 0
        
        composeTestRule.setContent {
            MaterialTheme {
                Column {
                    Text("Count: $count")
                    Button(onClick = { count++ }) {
                        Text("Tăng")
                    }
                }
            }
        }
        
        // Perform click
        composeTestRule.onNodeWithText("Tăng").performClick()
        
        // Verify
        composeTestRule.onNodeWithText("Count: 1").assertExists()
    }
    
    @Test
    fun `text input works correctly`() {
        var text = ""
        
        composeTestRule.setContent {
            MaterialTheme {
                TextField(
                    value = text,
                    onValueChange = { text = it },
                    label = { Text("Nhập") }
                )
            }
        }
        
        // Perform text input
        composeTestRule.onNodeWithText("Nhập")
            .performTextInput("Xin chào")
        
        // Verify
        composeTestRule.onNodeWithText("Xin chào").assertExists()
    }
}
```

### Idling Resources cho Async

```kotlin
class AsyncTest {
    
    @get:Rule
    val composeTestRule = createComposeRule()
    
    @Test
    fun `loading indicator disappears after data loads`() {
        composeTestRule.setContent {
            var isLoading by remember { mutableStateOf(true) }
            
            LaunchedEffect(Unit) {
                delay(1000)
                isLoading = false
            }
            
            if (isLoading) {
                CircularProgressIndicator()
            } else {
                Text("Data loaded")
            }
        }
        
        // Wait for async operation
        composeTestRule.waitUntil(timeoutMillis = 5000) {
            composeTestRule
                .onAllNodesWithText("Data loaded")
                .fetchSemanticsNodes().isNotEmpty()
        }
        
        composeTestRule.onNodeWithText("Data loaded").assertExists()
    }
}
```

```kotlin
fun SemanticsNodeInteraction.assertTextEquals(expected: String): SemanticsNodeInteraction {
    assertEquals(expected, fetchSemanticsNode().config[SemanticsProperties.Text].first().text)
    return this
}

fun SemanticsNodeInteraction.performTextInputClear(): SemanticsNodeInteraction {
    performTextInput("")
    return this
}

// Fake cho testing
class FakeUserRepository : UserRepository {
    private val users = mutableListOf<User>()
    
    override fun getUsers() = flowOf(Resource.Success(users.toList()))
    override suspend fun getUserById(id: Int) = Resource.Success(users.find { it.id == id })
    override suspend fun createUser(user: User): Resource<User> {
        users.add(user)
        return Resource.Success(user)
    }
}
```

### Mẹo thực tế

> **💡 Tip**: Sử dụng `testTag` để đánh dấu các composables quan trọng. Điều này giúp test dễ dàng hơn và không bị ảnh hưởng khi thay đổi text.

> **⚠️ Lưu ý**: Tránh test implementation details. Test hành vi (behavior) thay vì cách UI được render.

> **🎯 Best Practice**: Viết UI test cho các user flow quan trọng: Login, Register, Main feature. Không cần test mọi UI state.
