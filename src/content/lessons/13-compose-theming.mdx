## Theme, Color, Typography

### MaterialTheme trong Compose

MaterialTheme cung cấp hệ thống design nhất quán cho toàn bộ ứng dụng:

```kotlin
import androidx.compose.material3.*
import androidx.compose.runtime.Composable

@Composable
fun UngDungCuaToi(
    darkTheme: Boolean = isSystemInDarkTheme(),
    content: @Composable () -> Unit
) {
    MaterialTheme(
        colorScheme = if (darkTheme) darkColorScheme() else lightColorScheme(),
        typography = Typography(),
        content = content
    )
}
```

### Custom Theme

```kotlin
// Theme.kt
import androidx.compose.foundation.isSystemInDarkTheme
import androidx.compose.material3.*
import androidx.compose.runtime.Composable

private val LightColors = lightColorScheme(
    primary = Color(0xFF1976D2),
    onPrimary = Color.White,
    primaryContainer = Color(0xFFD3E3FD),
    secondary = Color(0xFF52575D),
    background = Color(0xFFFAFBFC),
    surface = Color.White,
)

private val DarkColors = darkColorScheme(
    primary = Color(0xFF90CAF9),
    onPrimary = Color(0xFF003258),
    primaryContainer = Color(0xFF00497D),
    secondary = Color(0xFFBBC7CE),
    background = Color(0xFF1A1C1E),
    surface = Color(0xFF1A1C1E),
)

@Composable
fun AppTheme(
    darkTheme: Boolean = isSystemInDarkTheme(),
    dynamicColor: Boolean = true, // Material You - Android 12+
    content: @Composable () -> Unit
) {
    val colorScheme = when {
        // Dynamic color cho Android 12+
        dynamicColor && Build.VERSION.SDK_INT >= Build.VERSION_CODES.S -> {
            val context = LocalContext.current
            if (darkTheme) dynamicDarkColorScheme(context)
            else dynamicLightColorScheme(context)
        }
        darkTheme -> DarkColors
        else -> LightColors
    }
    
    MaterialTheme(
        colorScheme = colorScheme,
        typography = AppTypography,
        shapes = AppShapes,
        content = content
    )
}
```

### Sử dụng Theme trong UI

```kotlin
@Composable
fun ManHinhVoiTheme() {
    // Truy cập màu từ theme
    val colors = MaterialTheme.colorScheme
    
    // Truy cập typography từ theme
    val typography = MaterialTheme.typography
    
    Column(
        modifier = Modifier
            .fillMaxSize()
            .background(colors.background)
            .padding(16.dp)
    ) {
        // Text với typography style
        Text(
            text = "Tiêu đề chính",
            style = typography.headlineLarge,
            color = colors.primary
        )
        
        Text(
            text = "Nội dung mô tả",
            style = typography.bodyLarge,
            color = colors.onBackground
        )
        
        // Card tự động sử dụng theme colors
        Card(
            colors = CardDefaults.cardColors(
                containerColor = colors.surface
            )
        ) {
            Text(
                text = "Nội dung card",
                modifier = Modifier.padding(16.dp),
                color = colors.onSurface
            )
        }
        
        // Button với theme colors
        Button(
            onClick = { },
            colors = ButtonDefaults.buttonColors(
                containerColor = colors.primary
            )
        ) {
            Text("Nút chính", color = colors.onPrimary)
        }
    }
}
``

### Dark/Light Mode Toggle

```kotlin
// State management cho theme
class ThemeViewModel : ViewModel() {
    private val _isDarkMode = MutableStateFlow(false)
    val isDarkMode: StateFlow<Boolean> = _isDarkMode.asStateFlow()
    
    private val _useDynamicColor = MutableStateFlow(true)
    val useDynamicColor: StateFlow<Boolean> = _useDynamicColor.asStateFlow()
    
    fun toggleDarkMode() {
        _isDarkMode.value = !_isDarkMode.value
    }
    
    fun toggleDynamicColor() {
        _useDynamicColor.value = !_useDynamicColor.value
    }
}

@Composable
fun ManHinhCaiDat(viewModel: ThemeViewModel = viewModel()) {
    val isDarkMode by viewModel.isDarkMode.collectAsState()
    val useDynamicColor by viewModel.useDynamicColor.collectAsState()
    
    Column {
        Row(
            modifier = Modifier
                .fillMaxWidth()
                .padding(16.dp),
            horizontalArrangement = Arrangement.SpaceBetween
        ) {
            Text("Chế độ tối")
            Switch(
                checked = isDarkMode,
                onCheckedChange = { viewModel.toggleDarkMode() }
            )
        }
        
        Row(
            modifier = Modifier
                .fillMaxWidth()
                .padding(16.dp),
            horizontalArrangement = Arrangement.SpaceBetween
        ) {
            Text("Màu động (Android 12+)")
            Switch(
                checked = useDynamicColor,
                onCheckedChange = { viewModel.toggleDynamicColor() }
            )
        }
    }
}
```

### Mẹo thực tế

> **💡 Tip**: Luôn sử dụng `MaterialTheme.colorScheme` thay vì hardcode màu. Điều này đảm bảo UI nhất quán và hỗ trợ dark mode tự động.

> **⚠️ Lưu ý**: Dynamic color chỉ hoạt động trên Android 12+. Đảm bảo cung cấp fallback color scheme cho các phiên bản cũ hơn.

> **🎯 Best Practice**: Tạo một Theme.kt file riêng chứa tất cả colors, typography, shapes. Điều này giúp dễ dàng maintain và thay đổi theme sau này.
