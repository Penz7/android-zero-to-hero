### Clean Code Practices

```kotlin
// ❌ Bad naming
fun proc(d: Data): Result {
    // process data
}

// ✅ Good naming
fun calculateMonthlyExpenses(transactions: List<Transaction>): MonthlyExpenseSummary {
    val groupedByMonth = transactions.groupBy { it.date.month }
    return MonthlyExpenseSummary(
        monthlyTotals = groupedByMonth.mapValues { (_, txns) ->
            txns.sumOf { it.amount }
        }
    )
}

// ❌ Too many parameters
fun createTransaction(
    amount: Double,
    category: String,
    date: Long,

### Documentation Best Practices

```kotlin
/**
 * Calculates the total expenses for a given month.
 *
 * This function filters transactions by the specified month and year,
 * then sums up all expense transactions (amounts < 0).
 *
 * @param transactions List of all transactions
 * @param month The month to calculate (1-12)
 * @param year The year to calculate
 * @return Total expenses as a positive Double, or 0.0 if no expenses
 * @throws IllegalArgumentException if month is not in 1-12 range
 *
 * @sample
 * ```
 * val transactions = listOf(
 *     Transaction(amount = -50.0, date = "2024-01-15"),
 *     Transaction(amount = -30.0, date = "2024-01-20")
 * )
 * val total = calculateMonthlyExpenses(transactions, 1, 2024)
 * // total = 80.0
 * ```
 */
fun calculateMonthlyExpenses(
    transactions: List<Transaction>,
    month: Int,
    year: Int
): Double {
    require(month in 1..12) { "Month must be between 1 and 12" }
    
    return transactions
        .filter { it.date.month == month && it.date.year == year }
        .filter { it.amount < 0 }
        .sumOf { abs(it.amount) }
}
```

> **⚠️ Lưu ý**: Đừng copy code từ tutorials. Recruiters có thể nhận ra và đánh giá thấp.
