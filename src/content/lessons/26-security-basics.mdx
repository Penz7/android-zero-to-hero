### EncryptedSharedPreferences

```kotlin
import androidx.security.crypto.EncryptedSharedPreferences
import androidx.security.crypto.MasterKeys

object SecureStorage {
    
    private val masterKeyAlias = MasterKeys.getOrCreate(MasterKeys.AES256_GCM_SPEC)
    
    private val encryptedPrefs by lazy {
        EncryptedSharedPreferences.create(
            "secure_prefs",
            masterKeyAlias,
            context,
            EncryptedSharedPreferences.PrefKeyEncryptionScheme.AES256_SIV,
            EncryptedSharedPreferences.PrefValueEncryptionScheme.AES256_GCM
        )
    }
    
    // Lưu token an toàn
    fun saveToken(token: String) {
        encryptedPrefs.edit().putString("auth_token", token).apply()
    }
    
    fun getToken(): String? {
        return encryptedPrefs.getString("auth_token", null)
    }
    
    fun clearToken() {
        encryptedPrefs.edit().remove("auth_token").apply()
    }
    
    // Lưu sensitive data
    fun saveUserData(key: String, value: String) {
        encryptedPrefs.edit().putString(key, value).apply()
    }
}
```

### Certificate Pinning với OkHttp

```kotlin
import okhttp3.CertificatePinner

object SecureNetwork {
    
    // Tạo certificate pinner
    private val certificatePinner = CertificatePinner.Builder()
        .add(
            "api.example.com",
            "sha256/AAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAAA=",
            "sha256/BBBBBBBBBBBBBBBBBBBBBBBBBBBBBBBBBBBBBBBBBBB="
        )
        .build()
    
    // OkHttp client với certificate pinning
    fun createSecureClient(): OkHttpClient {
        return OkHttpClient.Builder()
            .certificatePinner(certificatePinner)
            .connectTimeout(30, TimeUnit.SECONDS)
            .readTimeout(30, TimeUnit.SECONDS)
            .build()
    }
    
    // Lấy certificate hash
    // openssl s_client -connect api.example.com:443 | openssl x509 -pubkey -noout | openssl pkey -pubin -outform der | openssl dgst -sha256 -binary | openssl enc -base64
}
```

### Input Validation

```kotlin
// Validators
object Validators {
    
    fun isValidEmail(email: String): Boolean {
        return email.matches(
            "^[A-Za-z0-9+_.-]+@[A-Za-z0-9.-]+\\.[A-Za-z]{2,}$".toRegex()
        )
    }
    
    fun isValidPhone(phone: String): Boolean {
        return phone.matches("^0[0-9]{9}$".toRegex())
    }
    
    fun isValidPassword(password: String): Boolean {
        // Ít nhất 8 ký tự, 1 chữ hoa, 1 chữ thường, 1 số
        return password.matches(
            "^(?=.*[0-9])(?=.*[a-z])(?=.*[A-Z]).{8,}$".toRegex()
        )
    }
    
    fun sanitizeInput(input: String): String {
        return input
            .replace("<".toRegex(), "&lt;")
            .replace(">".toRegex(), "&gt;")
            .replace("\"".toRegex(), "&quot;")
            .replace("'".toRegex(), "&#x27;")
    }
}

// Usage trong ViewModel
class LoginViewModel : ViewModel() {
    
    fun login(email: String, password: String) {
        // Validate inputs
        if (!Validators.isValidEmail(email)) {
            _uiState.value = LoginUiState.Error("Email không hợp lệ")
            return
        }
        
        if (!Validators.isValidPassword(password)) {
            _uiState.value = LoginUiState.Error("Mật khẩu không đủ mạnh")
            return
        }
        
        // Proceed with login
    }
}
```

### Secure API Communication

```kotlin
// Thêm security headers
interface ApiService {
    
    @GET("users")
    suspend fun getUsers(
        @Header("Authorization") token: String,
        @Header("X-Request-ID") requestId: String = UUID.randomUUID().toString(),
        @Header("X-Timestamp") timestamp: String = System.currentTimeMillis().toString()
    ): List<User>
}

// Request signing
class SigningInterceptor : Interceptor {
    override fun intercept(chain: Interceptor.Chain): Response {
        val original = chain.request()
        val timestamp = System.currentTimeMillis().toString()
        val nonce = UUID.randomUUID().toString()
        
        // Tạo signature
        val signature = generateSignature(
            method = original.method,
            path = original.url.encodedPath,
            timestamp = timestamp,
            nonce = nonce,
            secret = BuildConfig.API_SECRET
        )
        
        // Thêm headers
        val signedRequest = original.newBuilder()
            .header("X-Timestamp", timestamp)
            .header("X-Nonce", nonce)
            .header("X-Signature", signature)
            .build()
        
        return chain.proceed(signedRequest)
    }
}
```

### Root Detection

```kotlin
object RootDetector {
    
    fun isDeviceRooted(): Boolean {
        return checkRootBinaries() || checkSuExists() || checkRootProperties()
    }
    
    private fun checkRootBinaries(): Boolean {
        val paths = arrayOf(
            "/system/app/Superuser.apk",
            "/sbin/su",
            "/system/bin/su",
            "/system/xbin/su",
            "/data/local/xbin/su",
            "/data/local/bin/su",
            "/system/sd/xbin/su",
            "/system/bin/failsafe/su",
            "/data/local/su"
        )
        return paths.any { File(it).exists() }
    }
    
    private fun checkSuExists(): Boolean {
        return try {
            Runtime.getRuntime().exec("which su")
            true
        } catch (e: Exception) {
            false
        }
    }
    
    private fun checkRootProperties(): Boolean {
        val properties = listOf(
            "ro.debuggable",
            "ro.secure"
        )
        return properties.any { prop ->
            getSystemProperty(prop) == "1"
        }
    }
}
```

### Mẹo thực tế

> **💡 Tip**: Luôn bật R8 cho release builds. Không chỉ bảo mật mà còn giảm kích thước APK.

> **⚠️ Lưu ý**: Không bao giờ lưu sensitive data (password, API keys) trong SharedPreferences thường. Sử dụng EncryptedSharedPreferences hoặc Android Keystore.

> **🎯 Best Practice**: Thực hiện security audit định kỳ. Sử dụng các công cụ như MobSF (Mobile Security Framework) để scan vulnerabilities.
